DROP TABLE IF EXISTS config
DROP TABLE IF EXISTS coordinates
DROP TABLE IF EXISTS segments
DROP TABLE IF EXISTS priority

CREATE TABLE config (_id bigserial primary key, global_max_id bigint DEFAULT 0, number_of_segments bigint DEFAULT 0, max_chain_id bigint DEFAULT 0)
CREATE TABLE coordinates (_id bigserial primary key, x integer, y integer, z integer, is_processing boolean DEFAULT False, processed boolean DEFAULT False, run_number bigint, chain_id bigint, prev_chain_idx int DEFAULT NULL, start_date timestamptz, end_date timestamptz DEFAULT NULL)
CREATE TABLE segments (_id bigserial primary key, segment_id bigint, size integer, x integer, y integer, z integer, chain_id bigint)
CREATE TABLE priority (_id bigserial primary key, x integer, y integer, z integer, force boolean, chain_id bigint, quality varchar, location varchar, prev_chain_idx bigint, processed boolean DEFAULT False)
